const int soundPin = A0; // Pin where the sound detector is connected
const float referenceVoltage = 5.0; // Reference voltage of the Arduino (5V for most boards)

void setup() {
  Serial.begin(9600); // Start the serial communication at 9600 baud rate
}

void loop() {
  int sensorValue = analogRead(soundPin); // Read the analog value from the sound detector
  float voltage = sensorValue * (referenceVoltage / 1023.0); // Convert the analog value to voltage

  Serial.print("Sound Detector Voltage: ");
  Serial.print(voltage);
  Serial.println(" V");

  delay(1000); // Wait for a second before the next reading
}
